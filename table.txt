DROP DATABASE db_main;
CREATE DATABASE db_main;
ALTER DATABASE db_main CHARACTER SET utf8mb4 COLLATE utf8mb4_bin;
USE db_main;

CREATE TABLE Users (
user_id INT NOT NULL AUTO_INCREMENT,
username varchar(255),
xp int,
coins int,
email varchar(255),
password varchar(255),
follows int,
followers int,
bio varchar(2000),
country varchar(255),
weight int,
age int,
energy int,
today_progress int,
week_progress int,
day varchar(200),
week varchar(200),
error_styles varchar(2000),
active_style varchar(200),
friends varchar(2000),
account_created DATE,
login_streak int,
last_login varchar(200),
logins int,
theme varchar(200),
privacy varchar(200),
PRIMARY KEY ( user_id )
);



CREATE TABLE Elements (
element_id INT NOT NULL AUTO_INCREMENT,
element_name varchar(255),
creator_name varchar(255),
creator_id varchar(255),
description varchar(2000),
duration varchar(255),
type varchar(255),
reports int,
element_usage int,
PRIMARY KEY ( element_id )
);

CREATE TABLE Comments(
comment_id INT NOT NULL AUTO_INCREMENT,
creator_name varchar(255),
creator_id varchar(255),
comment varchar(2000),
plan_id varchar(255),
reports int,
date varchar(255),
likes varchar(255),
PRIMARY KEY ( comment_id )
);



CREATE TABLE Plans (
plan_name varchar(255),
creator_name varchar(255),
creator_id varchar(255),
plan_id varchar(255),
plan_description varchar(2000),
duration int,
category varchar(255),
difficulty varchar(255),
reports int,
plan_usage int,
plan_stars int,
plan MEDIUMTEXT,
tags varchar(2000),
privacy int,
date int,
plan_views int,
PRIMARY KEY ( plan_name )
);



CREATE TABLE Follow(
id INT NOT NULL AUTO_INCREMENT,
follower_id varchar(255),
target_id varchar(255),
follower_name varchar(255),
target_name varchar(255),
PRIMARY KEY ( id )
);

CREATE TABLE Stars(
id INT NOT NULL AUTO_INCREMENT,
user_id varchar(255),
plan_id varchar(255),
username varchar(255),
plan_name varchar(2000),
PRIMARY KEY ( id )
);

CREATE TABLE Views(
id INT NOT NULL AUTO_INCREMENT,
user_id varchar(255),
plan_id varchar(255),
username varchar(255),
PRIMARY KEY ( id )
);


CREATE TABLE Live(
id INT NOT NULL AUTO_INCREMENT,
user_id varchar(255),
plan_id varchar(255),
username varchar(255),
plan_duration int,
plan_name varchar(2000),
active varchar(255),
started varchar(255),
profile varchar(255),
followOnly varchar(255),
passwordOnly varchar(255),
password varchar(255),
date int,
joined int,
PRIMARY KEY ( id )
);

CREATE TABLE Ip(
ip varchar(200),
user_id varchar(255),
reply MEDIUMTEXT,
last_reset int,
requests int,
banned BOOLEAN,
PRIMARY KEY ( ip )
);

CREATE TABLE Logins(
login_id INT NOT NULL AUTO_INCREMENT,
ip varchar(200),
ip_remote varchar(200),
user_id varchar(255),
country varchar(1000),
country_code varchar(255),
city varchar(1000),
latitude varchar(1000),
longitude varchar(1000),
asn varchar(1000),
org varchar(1000),
continent varchar(1000),
date int,
signup BOOLEAN,
PRIMARY KEY ( login_id )
);

CREATE TABLE Notifications(
notif_id INT NOT NULL AUTO_INCREMENT,
notification_text MEDIUMTEXT,
user_id varchar(255),
from_id varchar(255),
from_name varchar(255),
plan_id varchar(255),
viewed BOOLEAN,
date int,
type varchar(255),
PRIMARY KEY ( notif_id )
);
